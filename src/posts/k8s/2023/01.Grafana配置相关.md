> 修改配置文件（default.ini或者custom.ini）

```ini
[security]
## 可嵌入iframe
allow_embedding = true

## 匿名访问且配置角色
[auth.anonymous]
# 开启匿名访问
enabled = true

# 指定应用于未经身份验证的用户的组织名称
org_name = Main Org.

# 指定应用于未经身份验证的用户的角色
org_role = Admin
```
